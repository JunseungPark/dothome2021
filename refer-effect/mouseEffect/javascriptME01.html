window.addEventListener("mousemove", function(event){
    document.querySelector(".clientX").innerText = event.clientX + "px"; // 브라우저 X좌표 기준
    document.querySelector(".clientY").innerText = event.clientY + "px"; // 브라우저 Y좌표 기준
    document.querySelector(".offsetX").innerText = event.offsetX + "px"; // 요소 X좌표 기준
    document.querySelector(".offsetY").innerText = event.offsetY + "px"; // 요소 Y좌표 기준
    document.querySelector(".pageX").innerText = event.pageX + "px"; // 페이지 X좌표 기준
    document.querySelector(".pageY").innerText = event.pageY + "px"; // 페이지 Y좌표 기준
    document.querySelector(".screenX").innerText = event.screenX + "px"; // 디바이스 X좌표 기준
    document.querySelector(".screenY").innerText = event.screenY + "px"; // 디바이스 Y좌표 기준
});

window.addEventListener("mousemove", function(event){
    // document.querySelector(".cursor").style.left = event.clientX - 25 + "px";
    // document.querySelector(".cursor").style.top = event.clientY  - 25 + "px";

    let x = event.clientX - 25 + "px";
    let y = event.clientY - 25 + "px";
    document.querySelector(".cursor").style.cssText = "left:" + x + "; top:" + y;

});

let spanBox = document.querySelectorAll(".mouseCont span");
spanBox.forEach((ele, index) => {
    console.log(ele);
    ele.addEventListener("mouseover", function(){
        document.querySelector(".cursor").classList.add(`style${index + 1}`);
    })
    ele.addEventListener("mouseout", function(){
        document.querySelector(".cursor").classList.remove(`style${index + 1}`);
    })
});
