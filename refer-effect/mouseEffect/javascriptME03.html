const w = document.querySelector(".cursor").clientWidth / 2;
const h = document.querySelector(".cursor").clientHeight / 2;
document.addEventListener("mousemove", e => {
    gsap.to(".cursor",  {duration: 0.4, left:event.pageX - w, top:event.pageY - h});
})

const cursor = document.querySelector(".cursor");
const follower = document.querySelector(".cursor-follower");

document.querySelectorAll(".mouseCont em").forEach(em => {
    em.addEventListener("mouseenter", ()=>{
        cursor.classList.add("active");
    })
    em.addEventListener("mouseleave", ()=>{
        cursor.classList.remove("active");
    })
})

document.querySelectorAll(".info.left.list li").forEach(em => {
    em.addEventListener("click", function(){
        document.querySelectorAll(".info.left.list li").forEach(em => {
            em.classList.remove("active")
        })
        this.classList.add("active")
        const text = this.innerText;
        document.querySelector(".cursor").style.cssText = "mix-blend-mode:" + text;
    }) 
});
