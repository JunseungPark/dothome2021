function mouseMove(e) {
    let positionSlow = (e.pageX - (document.documentElement.clientWidth /2)) * 0.1;
    let positionFast = (e.pageX - (document.documentElement.clientWidth /2)) * 0.2;
    gsap.to(".spanSlow", {duration: 0.4, x:positionSlow})
    gsap.to(".spanFast", {duration: 0.4, x:-positionFast})
}
document.addEventListener("mousemove", mouseMove);
