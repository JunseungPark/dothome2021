const cursor = $(".cursor");
const follower = $(".cursor-follower");
$(window).mousemove(function(event){
    // cursor.css({left:event.pageX - 5, top:event.pageY - 5})
    // follower.css({left:event.pageX - 15, top:event.pageY - 15})
    gsap.to(cursor,  {duration: 0.3, left:event.pageX - 5, top:event.pageY - 5});
    gsap.to(follower,  {duration: 0.8, left:event.pageX - 15, top:event.pageY - 15});
});

$(".mouseCont em").hover(function(){
    cursor.addClass("active");
    follower.addClass("active");
},function(){
    cursor.removeClass("active");
    follower.removeClass("active");
});

$(".info").hover(function(){
    cursor.addClass("active");
    follower.addClass("active");
},function(){
    cursor.removeClass("active");
    follower.removeClass("active");
});

$(window).mousemove(function(event){
    $(".pageX").text(event.pageX) + "px"; // 페이지 X좌표 기준
    $(".pageY").text(event.pageY) + "px"; // 페이지 Y좌표 기준
});
