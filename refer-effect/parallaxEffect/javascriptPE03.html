window.addEventListener("scroll", () => {
    let scrollTop = document.documentElement.scrollTop || window.scrollY || window.pageYOffset;

    document.querySelector(".paraScroll span").innerHTML = parseInt(scrollTop); 


    // 메뉴숨기기
    // if (scrollTop > 10) {
    //     document.querySelector(".paraNav").classList.add("show");
    // } else {
    //     document.querySelector(".paraNav").classList.remove("show");
    // }

    document.querySelectorAll(".content-item").forEach((el, index) => {
        if (scrollTop > el.offsetTop -2 ) {
            document.querySelectorAll(".paraNav ul li ").forEach(el => {
                el.classList.remove("active");
            })
            document.querySelector(`.paraNav ul li:nth-child(${index + 1})`).classList.add("active")
        }
    });
})
