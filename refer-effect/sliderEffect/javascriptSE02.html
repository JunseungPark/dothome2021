const sliderWrap = document.querySelector(".slider-wrap");
const sliderImg = sliderWrap.querySelector(".slider-img");
const sliderInner = sliderImg.querySelector(".slider-inner");
const slider = sliderInner.querySelectorAll(".slider");

let currentIndex = 0;
let sliderCount = slider.length;
let sliderInterval = 2000;

setInterval(() => {
    currentIndex < sliderCount ? currentIndex++ : currentIndex = 0;

    let sliderPosition = currentIndex * -800;
    sliderInner.style.left = sliderPosition + "px";
    
}, sliderInterval);
