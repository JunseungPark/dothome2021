19. 콜백 함수 : 동기 / 비동기

{
    // 01

    function funA() {
        console.log("funA가 실행되었습니다.");
    }
    function funB() {
        console.log("funB가 실행되었습니다.");
    }
    funB();
    funA();

    // 02
    function funC() {
        // setTimeout(function(){
        //     console.log("funC가 실행되었습니다.");
        // }, 1000)

        setTimeout(() => {
            console.log("funC가 실행되었습니다.");
        }, 1000)
    }
    function funD() {
        console.log("funD가 실행되었습니다.");
    }
    funC();
    funD();

    // 03
    function funE(callback) {
        setTimeout(() => {
            console.log("funE가 실행되었습니다.")
            callback();
        }, 1000)
    }
    function funF() {
        console.log("funF가 실행되었습니다.");
    }
    funE(funF);
}

25. 함수 유형 : 함수와 객체(함수)를 이용한형태 

{
    const info = {
        num1: 1,
        name1: "박준승",
        job1: "웹퍼블리셔",
        num2: 2,
        name2: "박준순",
        job2: "프론트앤드 개발자",

        result1 : function() {
            document.write(info.num1 + ". 내 이름은 " + info.name1 + "이며, 직업은 " + info.job1 + "입니다.")
        },
        result2 : function() {
            document.write(info.num2 + ". 내 이름은 " + info.name2 + "이며, 직업은 " + info.job2 + "입니다.")
        }
    }

    info.result1();         
    info.result2();   
}

26. 함수 유형 : 객체 생성자 함수 

{
    function func(num, name, job) {
        this.num = num;
        this.name = name;
        this.job = job;

        this.result = function() {
                document.write(this.num + ". 내 이름은 " + this.name + "이며, 직업은 " + this.job + "입니다.")
        }
    }

    const info1 = new func("1", "박준승", "웹 퍼블리셔"); // 인스턴스 생성
    const info2 = new func("2", "박준순", "프론트앤드 개발자"); // 인스턴스 생성
    
    info1.result();
    info2.result();
}

}
